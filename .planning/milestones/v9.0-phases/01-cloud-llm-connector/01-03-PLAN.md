    The component has these structural sections, which this Task builds as the initial shell:

    ```javascript
    function LLMSettingsPanel(props) {
      const onBack = props.onBack;
      const onLogout = props.onLogout;

      // State
      const [settings, setSettings] = React.useState(null);
      const [loading, setLoading] = React.useState(true);
      const [saving, setSaving] = React.useState(false);
      const [error, setError] = React.useState(null);
      const [testResult, setTestResult] = React.useState(null);
      const [testInProgress, setTestInProgress] = React.useState(false);
      const [models, setModels] = React.useState([]);

      // Form state (for no-key / change mode)
      const [formProvider, setFormProvider] = React.useState("anthropic");
      const [formModel, setFormModel] = React.useState("");
      const [formApiKey, setFormApiKey] = React.useState("");
      const [formBaseUrl, setFormBaseUrl] = React.useState("");
      const [isChanging, setIsChanging] = React.useState(false);

      // Fetch settings on mount
      React.useEffect(function() {
        fetch("/llm/settings")
          .then(function(r) { return r.json(); })
          .then(function(data) {
            setSettings(data);
            setFormProvider(data.provider || "anthropic");
            setFormModel(data.model || "");
            if (data.baseUrl) setFormBaseUrl(data.baseUrl);
            // Fetch model list after settings
            return fetchModels(data.provider || "anthropic", data.apiKeyConfigured);
          })
          .catch(function(err) {
            setError("Could not load LLM settings.");
          })
          .finally(function() {
            setLoading(false);
          });
      }, []);

      // Render states...
      if (loading) {
        return e("div", { className: "llm-settings-page" },
          // Loading state
        );
      }

      // Determine state: no key vs active vs changing
      const hasKey = settings && settings.apiKeyConfigured;
      const showForm = !hasKey || isChanging;

      return e("div", { className: "llm-settings-page" },
        // Back button
        e("button", { className: "back-btn", onClick: onBack }, "< Back"),

        // Title
        e("h2", null, "LLM Settings"),

        // Status banner
        !hasKey
          ? e("div", { className: "llm-settings-banner warning" },
              "Using local Ollama (fallback)")
          : e("div", { className: "llm-settings-banner success" },
              "Connected: " + (settings.provider || "ollama")),
      );
    }
    ```
