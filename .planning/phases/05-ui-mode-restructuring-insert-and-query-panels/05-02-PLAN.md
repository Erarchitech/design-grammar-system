**1. Add `requestKnowledge` function** inside `GraphViewerPage`, after `submitKnowledgeIngest`:

```javascript
const requestKnowledge = async () => {
  if (!knowledgePromptText.trim()) { setKnowledgeStatus("Question is required"); return; }
  if (knowledgeIsRunning) return;
  setKnowledgeIsRunning(true);
  setKnowledgeResponseText("");
  setKnowledgeResponseCypher("");
  setKnowledgeProgress(0);
  setKnowledgeWorkflowStep(1);
  setKnowledgeStepTimes(Array(knowledgeQuerySteps.length).fill(null));
  setKnowledgeStatus("Sending question...");
  try {
    const cfg = window.GRAPH_CONFIG || {};
    const headers = { "Content-Type": "application/json" };
    if (cfg.n8nUser && cfg.n8nPassword) {
      headers["Authorization"] = "Basic " + btoa(cfg.n8nUser + ":" + cfg.n8nPassword);
    }
    const res = await fetch("/n8n/webhook/dg/knowledge-query", {
      method: "POST",
      headers: headers,
      body: JSON.stringify({ prompt: knowledgePromptText.trim(), project: project || "default-project" })
    });
    if (res.ok) {
      startKnowledgePollingLatest("knowledge-query", "knowledge-query");
    } else {
      setKnowledgeResponseText("Error: webhook returned " + res.status);
      setKnowledgeIsRunning(false);
    }
  } catch (err) {
    setKnowledgeResponseText("Error: " + err.message);
    setKnowledgeStatus("Failed");
    setKnowledgeIsRunning(false);
  }
};
```

Replace it with the full Query Knowledge panel (per D-12, mirrors Query Rules layout):

```javascript
knowledgeMode === "query"
  ? e(React.Fragment, null,
      // Prompt textarea (per D-12)
      e("textarea", {
        rows: 6,
        value: knowledgePromptText,
        className: "prompt-input",
        placeholder: "Ask a question about your project knowledge...",
        onChange: (ev) => setKnowledgePromptText(ev.target.value),
        style: { marginTop: "8px" }
      }),
      // Submit button
      e("button", {
        onClick: requestKnowledge,
        disabled: knowledgeIsRunning,
        style: { marginTop: "8px" }
      }, "Query Knowledge"),
      // Progress bar
      e("div", { className: "progress" },
        e("div", { style: { width: knowledgeProgress + "%" } })
      ),
      // Step indicators
      e("div", { className: "steps" },
        knowledgeQuerySteps.map((s, i) => {
          const elapsed = knowledgeStepTimes[i];
          const isActive = i + 1 === knowledgeStep;
          const live = isActive && knowledgeStepStartedAtRef.current
            ? roundSeconds(Date.now() - knowledgeStepStartedAtRef.current)
            : null;
          const suffix = elapsed !== null && elapsed !== undefined
            ? " (" + elapsed.toFixed(1) + "s)"
            : (live !== null ? " (" + live.toFixed(1) + "s)" : "");
          return e("div", { key: s, className: isActive ? "active" : "" }, s + suffix);
        })
      ),
      // Response textarea (NL answer)
      e("label", { style: { marginTop: "12px", display: "block" } }, "Response"),
      e("textarea", {
        className: "response-output",
        rows: 6,
        value: knowledgeResponseText,
        readOnly: true,
        placeholder: "Response will appear here..."
      }),
      // Workflow Cypher textarea (per QRYK-02 / D-12)
      e("div", null,
        e("label", null, "Workflow Cypher"),
        e("textarea", {
          className: "workflow-output",
          rows: 6,
          value: knowledgeResponseCypher,
          readOnly: true,
          placeholder: "Cypher used to query the knowledge graph will appear here..."
        })
      )
    )
  : null,
```
