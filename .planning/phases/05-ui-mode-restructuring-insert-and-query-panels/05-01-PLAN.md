```javascript
const clearKnowledgePoll = () => {
  if (knowledgePollRef.current) {
    clearTimeout(knowledgePollRef.current);
    knowledgePollRef.current = null;
  }
};

const setKnowledgeWorkflowStep = (nextStep) => {
  const now = Date.now();
  const currentStep = knowledgeStepRef.current;
  if (currentStep > 0 && knowledgeStepStartedAtRef.current) {
    const elapsed = roundSeconds(now - knowledgeStepStartedAtRef.current);
    setKnowledgeStepTimes((times) => {
      const next = times.length ? times.slice() : Array(knowledgeSteps.length).fill(null);
      next[currentStep - 1] = elapsed;
      return next;
    });
  }
  knowledgeStepRef.current = nextStep;
  setKnowledgeStep(nextStep);
  if (nextStep > 0) {
    knowledgeStepStartedAtRef.current = now;
  }
};

const finalizeKnowledgeWorkflowStep = () => {
  const now = Date.now();
  const currentStep = knowledgeStepRef.current;
  if (currentStep > 0 && knowledgeStepStartedAtRef.current) {
    const elapsed = roundSeconds(now - knowledgeStepStartedAtRef.current);
    setKnowledgeStepTimes((times) => {
      const next = times.length ? times.slice() : Array(knowledgeSteps.length).fill(null);
      next[currentStep - 1] = elapsed;
      return next;
    });
  }
};

const applyKnowledgeProgressUpdate = (data) => {
  const stepValue = data.step ?? data.payload?.step;
  if (typeof stepValue === "number" && stepValue > 0) {
    if (stepValue !== knowledgeStepRef.current) {
      setKnowledgeWorkflowStep(stepValue);
    }
  }
  const progressValue = data.progress ?? data.payload?.progress;
  if (typeof progressValue === "number" && Number.isFinite(progressValue)) {
    setKnowledgeProgress(Math.max(0, Math.min(100, progressValue)));
  }
  const message = data.message ?? data.payload?.message;
  if (typeof message === "string" && message.trim()) {
    setKnowledgeStatus(message);
  } else if (data.status === "running") {
    setKnowledgeStatus("Workflow running...");
  }
};

const startKnowledgePollingLatest = (workflowKey, modeType) => {
  const cfg = window.GRAPH_CONFIG || {};
  const base = normalizeBaseUrl(cfg.dataServiceUrl || "/data-service");
  const poll = async () => {
    try {
      const res = await fetch(base + "/execution-result/latest/" + workflowKey);
      if (!res.ok) {
        setKnowledgeStatus("Polling error: " + res.status);
      } else {
        const data = await res.json();
        const st = data?.status || "running";
        if (st !== "completed" && st !== "cancelled" && st !== "failed") {
          applyKnowledgeProgressUpdate(data);
        }
        if (st === "completed") {
          const payload = data?.payload || {};
          if (modeType === "knowledge-query") {
            setKnowledgeResponseText(payload.answer || payload.response || "");
            setKnowledgeResponseCypher(payload.cypher || "");
          } else {
            setKnowledgeResponseText("Created note: " + (payload.title || payload.note_title || payload.noteId || "unknown"));
          }
          clearKnowledgePoll();
          setKnowledgeWorkflowStep(5);
          setKnowledgeProgress(100);
          setKnowledgeStatus("Workflow completed");
          finalizeKnowledgeWorkflowStep();
          setKnowledgeIsRunning(false);
          return;
        }
        if (st === "cancelled") {
          setKnowledgeStatus("Workflow cancelled");
          finalizeKnowledgeWorkflowStep();
          setKnowledgeIsRunning(false);
          clearKnowledgePoll();
          return;
        }
        if (st === "failed") {
          setKnowledgeStatus("Workflow failed");
          setKnowledgeIsRunning(false);
          clearKnowledgePoll();
          return;
        }
        if (st === "unknown") {
          setKnowledgeStatus("Workflow running...");
        }
      }
    } catch (err) {
      setKnowledgeStatus("Polling failed: " + err.message);
    }
    knowledgePollRef.current = setTimeout(poll, 1500);
  };
  clearKnowledgePoll();
  poll();
};
```

const submitKnowledgeIngest = async () => {
  if (!knowledgePromptText.trim()) { setKnowledgeStatus("Prompt is required"); return; }
  if (knowledgeIsRunning) return;
  setKnowledgeIsRunning(true);
  setKnowledgeResponseText("");

      headers["Authorization"] = "Basic " + btoa(cfg.n8nUser + ":" + cfg.n8nPassword);

    const res = await fetch("/n8n/webhook/dg/knowledge-ingest", {

      headers: headers,

      setKnowledgeIsRunning(false);

    setKnowledgeResponseText("Error: " + err.message);
    setKnowledgeStatus("Failed");

};
